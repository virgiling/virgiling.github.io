# 任务看板

```dataview
TABLE WITHOUT ID 
	regexreplace(Tasks.text, "\[.*$", "") AS "任务", 
	default(object(
	    " ", "❌",
	    "/", "🔧",
	    "-", "⛓️‍💥", 
	    "x", "✅",
	    "*", "⭐")[Tasks.status], "?") AS "任务状态",
	Tasks.created AS "创建时间", 
	Tasks.due AS "截至日期", 
	Tasks.completion AS "完成日期", 
	regexreplace(Tasks.subtasks.text, "\[.*$", "") AS "子任务", 
	map(Tasks.subtasks.status, (s) => default(object(
	    " ", "❌",
	    "/", "🔧",
	    "-", "⛓️‍💥", 
	    "x", "✅",
	    "*", "⭐")[s], "?")) as "子任务状态"
FROM #task
FLATTEN file.tasks AS Tasks
WHERE file.tasks AND contains(Tasks.tags, "#task")
	AND meta(Tasks.header).subpath = "今日待办"
SORT default(Tasks.created, "") DESC
```
