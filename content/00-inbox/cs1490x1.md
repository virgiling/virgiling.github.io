第一个做法忽略了负载均衡，在这里我们考虑让每个线程一直运行，一次只渲染 `chunkSize` 行

```cpp
void workerThreadStart(WorkerArgs *const args) {
  int chunkSize = args->chunkSize;
  int startRows = args->threadId * chunkSize,
      epoch = chunkSize * args->numThreads;

  for (int i = 0; i < args->height / chunkSize / args->numThreads + 1; i++) {
    int numRows = chunkSize;
    if (startRows + numRows > args->height) {
      numRows = args->height - startRows;
    }
    mandelbrotSerial(args->x0, args->y0, args->x1, args->y1, args->width,
                     args->height, startRows, numRows, args->maxIterations,
                     args->output);
    startRows += epoch;
  }
}
```

我们通过 `bash` 脚本运行代码并统计数据，通过 `python` 来绘制图片

```bash
#!/bin/bash

VIEW=1

while [[ "$1" != "" ]]; do
    case $1 in
        -v | --view )          VIEW="$2"
                               shift
                               ;;
        * )                    echo "Invalid option: $1"
                               exit 1
    esac
    shift
done


CSV_FILE="mandelbrot_results_load_balance_v"${VIEW}_".csv"
echo "Threads,SerialTime(ms),ThreadTime(ms),Speedup" > "$CSV_FILE"

make clean && make -j

wait

for t in {2..16}; do
    echo "Testing $t threads..."
    output=$(taskset -c 0-7 ./mandelbrot -t $t -v 2 2>&1)
    wait

    serial=$(echo "$output" | grep "mandelbrot serial" | grep -oE '\[[0-9.]+\]' | tr -d '[]')
    thread=$(echo "$output" | grep "mandelbrot thread" | grep -oE '\[[0-9.]+\]' | tr -d '[]')
    speedup=$(echo "$output" | grep "speedup" | grep -oE '[0-9.]+x' | tr -d 'x')

    if [[ $serial && $thread && $speedup ]]; then
        echo "$t,$serial,$thread,$speedup" >> "$CSV_FILE"
        echo "  $t threads: serial=${serial}ms, thread=${thread}ms, speedup=${speedup}x"
    fi
done

echo "Done! Results in $CSV_FILE"
```
